2. 函数传参: `void func(int arr[])` 或 `(int *arr)`，一般还要传入数组长度`int n`。二维数组用`int arr[][N]`或`int (*arr)[N]`，其中`N`是列数。
3. 插入删除
```c++
// 删除指定位置的元素，调用示例代码：deleteElement(arr, &n, index);
void deleteElement(int arr[], int *n, int index) {
    if(index < 0 || index >= *n) return;
    for(int i = index; i < *n-1; i++)
        arr[i] = arr[i+1];
    (*n)--;
}
// 在指定位置插入元素
void insertElement(int arr[], int *n, int index, int value) {
    if(index < 0 || index > *n) return;
    for(int i = *n; i > index; i--)
        arr[i] = arr[i-1];
    arr[index] = value;
    (*n)++;
}
```

### 静态分配的顺序表
```c++
#include <iostream>
using namespace std;

const int MAXSIZE = 100;  // 定义顺序表的最大长度

typedef struct {
    int data[MAXSIZE];  // 存储数据的数组
    int length;         // 当前长度
} SqList;

// 初始化顺序表
void InitList(SqList &L) {
    L.length = 0;
}

// 顺序输出顺序表
void PrintList(SqList L) {
    if (L.length == 0) {
        cout << "顺序表为空" << endl;
        return;
    }
    cout << "顺序表元素：";
    for (int i = 0; i < L.length; i++)
        cout << L.data[i] << " ";
    cout << endl;
}

// 在位置i插入元素e
bool ListInsert(SqList &L, int i, int e) {
    if (i < 1 || i > L.length + 1) {  // 判断i的范围是否有效
        cout << "插入位置不合法" << endl;
        return false;
    }
    if (L.length >= MAXSIZE) {  // 判断顺序表是否已满
        cout << "顺序表已满，无法插入" << endl;
        return false;
    }
    for (int j = L.length; j >= i; j--)  // 将第i个元素及之后的元素后移。始于L.length，终于i
        L.data[j] = L.data[j - 1];
    L.data[i - 1] = e;  // 在位置i处放入e
    L.length++;         // 别忘
    return true;
}

// 删除位置i的元素，并用e返回其值
bool ListDelete(SqList &L, int i, int &e) {
    if (i < 1 || i > L.length) {  // 判断i的范围是否有效
        cout << "删除位置不合法" << endl;
        return false;
    }
    if (L.length == 0) {  // 判断顺序表是否为空
        cout << "顺序表为空，无法删除" << endl;
        return false;
    }
    e = L.data[i - 1];  // 将被删除的元素赋值给e
    for (int j = i; j < L.length; j++)   // 将第i个位置后的元素前移。始于i，终于L.length-1
        L.data[j - 1] = L.data[j];
    L.length--;  // 别忘
    return true;
}

// 按值查找，返回元素e的位置（从1开始）
int LocateElem(SqList L, int e) {
    for (int i = 0; i < L.length; i++)
        if (L.data[i] == e)
            return i + 1;  // 返回顺序表中的位置（从1开始）
    return 0;  // 查找失败，返回0
}

// 获取位置i的元素，并用e返回其值
bool GetElem(SqList L, int i, int &e) {
    if (i < 1 || i > L.length) {
        cout << "位置不合法" << endl;
        return false;
    }
    e = L.data[i - 1];
    return true;
}

int main() {
    SqList L;
    InitList(L);
    // 插入元素
    for (int i = 1; i <= 5; i++)
        ListInsert(L, i, i * 10);
    PrintList(L);  // 顺序表元素：10 20 30 40 50
    // 在指定位置插入元素
    ListInsert(L, 3, 99);
    PrintList(L);  // 顺序表元素：10 20 99 30 40 50
    // 删除元素
    int deletedElem;
    if (ListDelete(L, 4, deletedElem))
        cout << "删除的元素是：" << deletedElem << endl;  // 删除的元素是：30
    PrintList(L);  // 顺序表元素：10 20 99 40 50
    // 查找元素
    int pos = LocateElem(L, 99);
    if (pos > 0)
        cout << "元素99的位置是：" << pos << endl;  // 元素99的位置是：3
    else
        cout << "未找到元素99" << endl;
    // 获取元素
    int elem;
    if (GetElem(L, 2, elem))
        cout << "第二个元素是：" << elem << endl;  // 第二个元素是：20
    return 0;
}
```

### 动态分配的顺序表
```c++
#include <iostream>
using namespace std;

#define INIT_SIZE 10  // 初始容量

typedef struct {
    int *data;      // 指向动态分配数组的指针
    int maxsize;    // 当前最大容量
    int length;     // 当前长度
} SqList;

// 初始化顺序表
void InitList(SqList &L) {
    L.data = new int[INIT_SIZE];  // L->data = (int *) malloc(INIT_SIZE * sizeof(int));
    if (!L.data) {
        cout << "内存分配失败" << endl;
        exit(1);
    }
    L.maxsize = INIT_SIZE;
    L.length = 0;
}

// 扩容函数
void IncreaseSize(SqList &L, int len) {
    int *p = L.data;
    L.data = new int[L.maxsize + len];  // 重新分配更大的空间
    if (!L.data) {
        cout << "内存分配失败" << endl;
        exit(1);
    }
    // 复制原有数据到新空间
    for (int i = 0; i < L.length; i++)
        L.data[i] = p[i];
    // 以上相当于(省略new_data == NULL的判断)
    // int *new_data = (int *) realloc(L->data, (L->maxsize + len) * sizeof(int));
    // L->data = new_data;
    L.maxsize += len;  // 更新最大容量
    delete[] p;        // 释放原有空间
    cout << "顺序表已扩容，新容量为: " << L.maxsize << endl;
}

// 自动扩容检查
void CheckCapacity(SqList &L) {
    if (L.length >= L.maxsize) {
        // 当元素个数达到最大容量时，自动扩容
        IncreaseSize(L, L.maxsize);  // 扩容为原来的两倍
    }
}

// 顺序输出顺序表
void PrintList(SqList L) {
    if (L.length == 0) {
        cout << "顺序表为空" << endl;
        return;
    }
    cout << "顺序表元素（容量" << L.maxsize << "，长度" << L.length << "）: ";
    for (int i = 0; i < L.length; i++)
        cout << L.data[i] << " ";
    cout << endl;
}

// 在位置i插入元素e
bool ListInsert(SqList &L, int i, int e) {
    if (i < 1 || i > L.length + 1) {  // 判断i的范围是否有效
        cout << "插入位置不合法" << endl;
        return false;
    }
    CheckCapacity(L);  // 检查是否需要扩容
    for (int j = L.length; j >= i; j--) {  // 将第i个元素及之后的元素后移
        L.data[j] = L.data[j - 1];
    }
    L.data[i - 1] = e;  // 在位置i处放入e
    L.length++;         // 长度加1
    return true;
}

// 删除位置i的元素，并用e返回其值
// C用int ListDelete(SqList *L, int i, int *e) 与 ListDelete(&L, 5, &e)调用;
bool ListDelete(SqList &L, int i, int &e) {
    if (i < 1 || i > L.length) {  // 判断i的范围是否有效
        cout << "删除位置不合法" << endl;
        return false;
    }
    if (L.length == 0) {  // 判断顺序表是否为空
        cout << "顺序表为空，无法删除" << endl;
        return false;
    }
    e = L.data[i - 1];  // 将被删除的元素赋值给e。相当于 *e = L->data[i - 1];
    for (int j = i; j < L.length; j++)  // 将第i个位置后的元素前移
        L.data[j - 1] = L.data[j];
    L.length--;  // 长度减1
    return true;
}

// 按值查找，返回元素e的位置（从1开始）
int LocateElem(SqList L, int e) {
    for (int i = 0; i < L.length; i++)
        if (L.data[i] == e)
            return i + 1;  // 返回位置（从1开始）
    return 0;  // 查找失败，返回0
}

// 获取位置i的元素，并用e返回其值
// C用int GetElem(SqList *L, int i, int *e) 与 GetElem(&L, 3, &e);调用
bool GetElem(SqList L, int i, int &e) {
    if (i < 1 || i > L.length) {
        cout << "位置不合法" << endl;
        return false;
    }
    e = L.data[i - 1];  // 相当于*e = L->data[i - 1];将获取的元素赋值给e指向的位置
    return true;
}

// 销毁顺序表，释放内存
void DestroyList(SqList &L) {
    if (L.data) {
        delete[] L.data;  // 相当于 free(L->data);
        L.data = nullptr;  // 相当于 L->data = NULL;
    }
    L.maxsize = 0;
    L.length = 0;
    cout << "顺序表已销毁" << endl;
}

int main() {
    SqList L;
    InitList(L);
    // 插入元素，测试自动扩容
    for (int i = 1; i <= 15; i++) {
        ListInsert(L, i, i * 10);  // 插入10,20,...,150
    }
    PrintList(L);  // 容量20，长度15（省略了部分输出，下同）
    // 手动扩容
    IncreaseSize(L, 10);
    PrintList(L);  // 容量30，长度15
    // 继续插入元素
    for (int i = 16; i <= 25; i++) {
        ListInsert(L, i, i * 10);
    }
    PrintList(L);  // 容量30，长度25
    // 删除元素
    int deletedElem;
    if (ListDelete(L, 5, deletedElem))
        cout << "删除的元素是：" << deletedElem << endl;  // 删除的元素是：50
    PrintList(L);  // 容量30，长度24
    // 查找元素
    int pos = LocateElem(L, 150);
    if (pos > 0)
        cout << "元素150的位置是：" << pos << endl;  // 元素150的位置是：14
    // 获取元素
    int elem;
    if (GetElem(L, 10, elem))
        cout << "第10个元素是：" << elem << endl;  // 第10个元素是：110
    // 销毁顺序表
    DestroyList(L);
    return 0;
}
```

### KMP
```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

/*
 KMP 实现（0-based，next[0] = -1 风格）
 提供：
  - compute_next     : 计算经典 next（失配时跳转位置）
  - compute_next_val : 在 next 基础上生成 next_val（优化版）
  - kmp_search       : 使用给定 next 数组在文本中查找模式，返回首次匹配的下标（0-based），找不到返回 -1
*/

/* 计算经典 next 数组（next[0] = -1） 
   p : 模式串，长度 m
   next : 外部分配长度至少为 m 的数组
*/
void compute_next(const char *p, int m, int next[]) {
    int i = 0, j = -1;
    next[0] = -1;
    while (i < m - 1) {
        if (j == -1 || p[i] == p[j]) {
            ++i; ++j;
            next[i] = j;
        } else {
            j = next[j];
        }
    }
}

/* 基于经典 next 生成 next_val（优化版）
   将会修改 next，使得在 p[i] == p[next[i]] 的情况下把 next[i] 跳过为 next[next[i]]
   使用方法：先调用 compute_next，再调用 compute_next_val（两者都需要 next 数组）
*/
void compute_next_val(const char *p, int m, int next[]) {
    // 假设 next 已由 compute_next 填好
    for (int i = 1; i < m; ++i) {
        if (next[i] != -1 && p[i] == p[next[i]]) {
            next[i] = next[next[i]];  // 跳过等于的情况
        }
    }
}

/* KMP 搜索（使用提供的 next 数组）
   s : 文本串，长度 n
   p : 模式串，长度 m
   next : next 数组（经典或优化版均可），长度至少为 m
   返回值：首次匹配的起始下标（0-based），未找到返回 -1
*/
int kmp_search(const char *s, const char *p, const int next[]) {
    int n = (int)strlen(s);
    int m = (int)strlen(p);
    if (m == 0) return 0; // 空模式约定为匹配位置 0

    int i = 0, j = 0;
    while (i < n && j < m) {
        if (j == -1 || s[i] == p[j]) {
            ++i; ++j;
        } else {
            j = next[j];
        }
    }
    if (j == m) return i - j;
    return -1;
}

/* 示例用法 */
int main() {
    const char *text = "ababcabcacbab";
    const char *pattern = "abcac";

    int m = (int)strlen(pattern);
    if (m == 0) {
        printf("空模式，返回 0\n");
        return 0;
    }

    // 分配 next 数组
    int *next1 = (int *)malloc(sizeof(int) * m);
    if (!next1) return -1;

    // 1) 经典 next
    compute_next(pattern, m, next1);
    printf("classic next: ");
    for (int i = 0; i < m; ++i) printf("%d ", next1[i]);
    printf("\n");

    int pos1 = kmp_search(text, pattern, next1);
    printf("使用 classic next，匹配位置: %d\n", pos1);

    // 2) next_val（优化版）
    int *next_val = (int *)malloc(sizeof(int) * m);
    if (!next_val) {
        free(next1);
        return -1;
    }
    // 先拷贝经典 next，再做优化（也可直接在 next1 上修改）
    for (int i = 0; i < m; ++i) next_val[i] = next1[i];
    compute_next_val(pattern, m, next_val);

    printf("optimized next_val: ");
    for (int i = 0; i < m; ++i) printf("%d ", next_val[i]);
    printf("\n");

    int pos2 = kmp_search(text, pattern, next_val);
    printf("使用 next_val，匹配位置: %d\n", pos2);

    free(next1);
    free(next_val);
    return 0;
}
```

```c
/* =========================
   1) 二叉树直径（返回节点数的直径）   复杂度：O(n)
   方法：单次 DFS 返回子树高度 & 更新全局直径
   ========================= */
int diameter_dfs(TreeNode* u, int &best) {
    if(!u) return 0;
    int lh = diameter_dfs(u->l, best);
    int rh = diameter_dfs(u->r, best);
    best = max(best, lh + rh + 1); // 节点数直径；若要边数，使用 lh+rh
    return 1 + max(lh, rh);
}
int tree_diameter(TreeNode* root) {
    int best = 0;
    diameter_dfs(root, best);
    return best;
}

/* =========================
   4) 判断是否为高度平衡二叉树（AVL 定义：任意节点左右子树高度差 ≤ 1）   复杂度：O(n)
   方法：后序计算高度并传回是否平衡
   ========================= */
pair<int,bool> balanced_dfs(TreeNode* u) {
    if(!u) return {0, true};
    auto L = balanced_dfs(u->l);
    auto R = balanced_dfs(u->r);
    int h = 1 + max(L.first, R.first);
    bool ok = L.second && R.second && (abs(L.first - R.first) <= 1);
    return {h, ok};
}
bool is_balanced(TreeNode* root) {
    return balanced_dfs(root).second;
}

/* =========================
   5) 二叉树最大路径和（路径可从任意节点到任意节点，至少包含一个节点）   复杂度：O(n)
   方法：递归维护通过当前节点的最大单侧路径，并更新全局最大
   ========================= */
int maxPathSum_dfs(TreeNode* u, int &best) {
    if(!u) return INT_MIN/2; // 避免溢出
    int L = maxPathSum_dfs(u->l, best);
    int R = maxPathSum_dfs(u->r, best);
    int single = max({u->val, u->val + L, u->val + R}); // 单侧路径（供父节点使用）
    int through = max(single, u->val + L + R); // 经过当前节点的最大
    best = max(best, through);
    return single;
}
int maxPathSum(TreeNode* root) {
    if(!root) return 0;
    int best = INT_MIN;
    maxPathSum_dfs(root, best);
    return best;
}

/* =========================
   6) Huffman 最小 WPL（给定若干权值，构造 Huffman 树的最小加权路径和）   复杂度：O(m log m)
   方法：优先队列贪心合并
   ========================= */
long long huffman_wpl(const vector<long long>& weights) {
    priority_queue<long long, vector<long long>, greater<long long>> pq;
    for(auto w: weights) if(w>0) pq.push(w);
    if(pq.empty()) return 0;
    long long ans = 0;
    while(pq.size() >= 2) {
        long long a = pq.top(); pq.pop();
        long long b = pq.top(); pq.pop();
        long long s = a + b;
        ans += s;
        pq.push(s);
    }
    return ans;
}

/* =========================
   7) 判断两棵二叉树是否同构（结构和值完全相同）   复杂度：O(n)
   也可扩展为：判断是否镜像同构（如允许左右交换）
   ========================= */
bool same_tree(TreeNode* a, TreeNode* b) {
    if(!a && !b) return true;
    if(!a || !b) return false;
    if(a->val != b->val) return false;
    return same_tree(a->l, b->l) && same_tree(a->r, b->r);
}
// 允许左右子树交换（树的同构 / isomorphism with swapping）
bool isomorphic_with_swap(TreeNode* a, TreeNode* b) {
    if(!a && !b) return true;
    if(!a || !b) return false;
    if(a->val != b->val) return false;
    // either not swapped, or swapped
    return (isomorphic_with_swap(a->l, b->l) && isomorphic_with_swap(a->r, b->r))
        || (isomorphic_with_swap(a->l, b->r) && isomorphic_with_swap(a->r, b->l));
}
```

```c
#include <stdio.h>

int qempty(Queue *Q){ return Q->head == Q->tail; }
void qpush(Queue *Q, int x){ Q->q[Q->tail++] = x; if(Q->tail==MAXN) Q->tail = 0; }
int qpop(Queue *Q){ int x = Q->q[Q->head++]; if(Q->head==MAXN) Q->head = 0; return x; }

/* =========================
   1) BFS（无权图，邻接矩阵）
   - 参数: n, adj (0/1), s 起点
   - 输出: dist[] (距离，-1 表示不可达), parent[] (-1 表示无父)
   ========================= */
void BFS_matrix(int n, int adj[MAXN][MAXN], int s, int dist[MAXN], int parent[MAXN]) {
    int i;
    for(i=0;i<n;i++){ dist[i] = -1; parent[i] = -1; }
    Queue Q; qinit(&Q);
    dist[s] = 0; parent[s] = -1;
    qpush(&Q, s);
    while(!qempty(&Q)){
        int u = qpop(&Q);
        for(int v=0; v<n; ++v){
            if(adj[u][v] && dist[v] == -1){
                dist[v] = dist[u] + 1;
                parent[v] = u;
                qpush(&Q, v);
            }
        }
    }
}

/* =========================
   2) DFS（递归，邻接矩阵）
   - 参数: n, adj (0/1), s 起点
   - 输出: visited[], parent[], tin/tout 或 depth 可扩展
   ========================= */
void DFS_matrix_rec(int n, int adj[MAXN][MAXN], int u, int visited[MAXN], int parent[MAXN]) {
    visited[u] = 1;
    for(int v=0; v<n; ++v){
        if(adj[u][v] && !visited[v]){
            parent[v] = u;
            DFS_matrix_rec(n, adj, v, visited, parent);
        }
    }
}
/* 便捷包装 */
void DFS_matrix(int n, int adj[MAXN][MAXN], int s, int visited[MAXN], int parent[MAXN]) {
    int i;
    for(i=0;i<n;i++){ visited[i]=0; parent[i]=-1; }
    parent[s] = -1;
    DFS_matrix_rec(n, adj, s, visited, parent);
}

/* =========================
   3) 拓扑排序（邻接矩阵，Kahn 算法）
   - 参数: n, adj (有向图, 0/1)
   - 输出: order[] 返回拓扑序列长度 len；若有环返回 0 并 len=0
   ========================= */
int topo_kahn_matrix(int n, int adj[MAXN][MAXN], int order[MAXN]) {
    int indeg[MAXN]; memset(indeg,0,sizeof(indeg));
    for(int u=0; u<n; ++u)
        for(int v=0; v<n; ++v)
            if(adj[u][v]) indeg[v]++;
    Queue Q; qinit(&Q);
    for(int i=0;i<n;i++) if(indeg[i]==0) qpush(&Q, i);
    int idx = 0;
    while(!qempty(&Q)){
        int u = qpop(&Q);
        order[idx++] = u;
        for(int v=0; v<n; ++v){
            if(adj[u][v]){
                indeg[v]--;
                if(indeg[v]==0) qpush(&Q, v);
            }
        }
    }
    if(idx < n){ /* 有环 */
        return 0;
    }
    return idx;
}

/* =========================
   4) Dijkstra（邻接矩阵，非负权）
   - 参数: n, adj (带权，W_INF 表示无边), s
   - 输出: dist[] (ll), parent[]
   - 复杂度：O(n^2)
   ========================= */
void dijkstra_matrix(int n, int adj[MAXN][MAXN], int s, ll dist[MAXN], int parent[MAXN]) {
    int vis[MAXN]; memset(vis,0,sizeof(vis));
    for(int i=0;i<n;i++){
        dist[i] = (ll)W_INF;
        parent[i] = -1;
    }
    dist[s] = 0;
    for(int it=0; it<n; ++it){
        int u = -1;
        ll best = (ll)W_INF;
        for(int i=0;i<n;i++){
            if(!vis[i] && dist[i] < best){ best = dist[i]; u = i; }
        }
        if(u == -1) break;
        vis[u] = 1;
        for(int v=0; v<n; ++v){
            if(adj[u][v] != W_INF && !vis[v]){
                ll nd = dist[u] + (ll)adj[u][v];
                if(nd < dist[v]){
                    dist[v] = nd;
                    parent[v] = u;
                }
            }
        }
    }
}

/* =========================
   5) Bellman-Ford（邻接矩阵实现）
   - 参数: n, adj (带权，W_INF 表示无边), s
   - 输出: dist[], parent[]；返回 1 表示无负环，0 表示存在负环可到达
   - 复杂度：O(n^3)（因为每次松弛需 O(n^2) 扫描矩阵）
   ========================= */
int bellman_ford_matrix(int n, int adj[MAXN][MAXN], int s, ll dist[MAXN], int parent[MAXN]) {
    for(int i=0;i<n;i++){ dist[i] = (ll)W_INF; parent[i] = -1; }
    dist[s] = 0;
    for(int it=1; it<=n-1; ++it){
        int changed = 0;
        for(int u=0; u<n; ++u){
            if(dist[u] == (ll)W_INF) continue;
            for(int v=0; v<n; ++v){
                if(adj[u][v] == W_INF) continue;
                ll nd = dist[u] + (ll)adj[u][v];
                if(nd < dist[v]){
                    dist[v] = nd;
                    parent[v] = u;
                    changed = 1;
                }
            }
        }
        if(!changed) break;
    }
    /* 检测负环 */
    for(int u=0; u<n; ++u){
        if(dist[u] == (ll)W_INF) continue;
        for(int v=0; v<n; ++v){
            if(adj[u][v] == W_INF) continue;
            if(dist[u] + (ll)adj[u][v] < dist[v]) return 0; /* 发现负环 */
        }
    }
    return 1;
}

/* =========================
   6) Kruskal（邻接矩阵 -> 边表 -> Kruskal）
   - 假定图为无向带权（对称矩阵），W_INF 表示无边
   - 输出：返回 MST 总权重，若不连通返回 -1
   ========================= */
typedef struct { int u,v,w; } Edge;
int cmpEdge(const void *a, const void *b){
    Edge *A = (Edge*)a; Edge *B = (Edge*)b;
    return (A->w > B->w) - (A->w < B->w);
}
int findp(int p[], int x){ return p[x]==x?x:(p[x]=findp(p,p[x])); }
int unitep(int p[], int r[], int a, int b){
    a = findp(p,a); b = findp(p,b);
    if(a==b) return 0;
    if(r[a] < r[b]) { p[a] = b; }
    else if(r[a] > r[b]) { p[b] = a; }
    else { p[b]=a; r[a]++; }
    return 1;
}
ll kruskal_matrix(int n, int adj[MAXN][MAXN]) {
    Edge edges[MAXN*MAXN];
    int m = 0;
    for(int u=0; u<n; ++u){
        for(int v=u+1; v<n; ++v){ /* 上三角避免重复 */
            if(adj[u][v] != W_INF){
                edges[m].u = u; edges[m].v = v; edges[m].w = adj[u][v];
                m++;
            }
        }
    }
    qsort(edges, m, sizeof(Edge), cmpEdge);
    int p[MAXN], rnk[MAXN];
    for(int i=0;i<n;i++){ p[i]=i; rnk[i]=0; }
    ll total = 0; int cnt = 0;
    for(int i=0;i<m && cnt < n-1; ++i){
        if(unitep(p, rnk, edges[i].u, edges[i].v)){
            total += edges[i].w; cnt++;
        }
    }
    if(cnt < n-1) return -1; /* 不连通 */
    return total;
}

/* =========================
   7) Prim（邻接矩阵，O(n^2) 版本）
   - 输入: n, adj (带权，W_INF 表示无边)
   - 输出: 返回 MST 权重；若不连通返回 -1
   ========================= */
ll prim_matrix(int n, int adj[MAXN][MAXN], int start) {
    int used[MAXN]; ll mincost[MAXN];
    for(int i=0;i<n;i++){ used[i]=0; mincost[i] = (ll)W_INF; }
    mincost[start] = 0;
    ll total = 0; int cnt = 0;
    for(int it=0; it<n; ++it){
        int u = -1;
        ll best = (ll)W_INF;
        for(int i=0;i<n;i++){
            if(!used[i] && mincost[i] < best){ best = mincost[i]; u = i; }
        }
        if(u == -1) break;
        used[u] = 1;
        total += mincost[u];
        cnt++;
        for(int v=0; v<n; ++v){
            if(!used[v] && adj[u][v] != W_INF && adj[u][v] < mincost[v]){
                mincost[v] = adj[u][v];
            }
        }
    }
    if(cnt < n) return -1;
    return total;
}

/* =========================
   8) 欧拉路径 / 欧拉回路（有向图，邻接矩阵）
   - 要求：给定有向图（可以推广到无向），返回是否存在欧拉路径并输出一条路径（顶点序列）
   - 算法：Hierholzer，使用矩阵时直接减少边计数 adj[u][v]--
   - 为保持原矩阵，函数内部会复制一份 work[u][v]
   ========================= */
int euler_directed_matrix(int n, int adj_in[MAXN][MAXN], int start, int path[], int *path_len) {
    /* 计算入度/出度 */
    int outdeg[MAXN], indeg[MAXN];
    int work[MAXN][MAXN];
    for(int i=0;i<n;i++){
        outdeg[i]=indeg[i]=0;
        for(int j=0;j<n;j++){
            work[i][j] = adj_in[i][j]; /* 支持多重边计数 */
            if(adj_in[i][j] != 0){
                outdeg[i] += adj_in[i][j];
                indeg[j] += adj_in[i][j];
            }
        }
    }
    /* 判断存在性：欧拉回路或路径条件 */
    int start_nodes = 0, end_nodes = 0;
    for(int i=0;i<n;i++){
        if(outdeg[i] - indeg[i] == 1) start_nodes++;
        else if(indeg[i] - outdeg[i] == 1) end_nodes++;
        else if(indeg[i] != outdeg[i] && !(indeg[i]==outdeg[i])) { /* 其余不匹配 */
            *path_len = 0; return 0;
        }
    }
    if(!((start_nodes==0 && end_nodes==0) || (start_nodes==1 && end_nodes==1))) {
        *path_len = 0; return 0;
    }
    /* Hierholzer（用栈） */
    int stack_v[MAXN*MAXN], top = 0;
    int res[MAXN*MAXN], rpos = 0;
    stack_v[top++] = start;
    while(top > 0){
        int u = stack_v[top-1];
        int v;
        for(v=0; v<n; ++v) if(work[u][v] > 0) break;
        if(v < n){ /* 有出边 */
            work[u][v]--;
            stack_v[top++] = v;
        } else { /* 无出边，回溯输出 */
            res[rpos++] = u;
            top--;
        }
    }
    /* res 是逆序顶点序列，长度 rpos */
    /* 验证是否使用了所有边：检查是否所有 work[u][v]==0 */
    for(int i=0;i<n;i++) for(int j=0;j<n;j++) if(work[i][j] != 0){ *path_len = 0; return 0; }
    /* 反转到 path */
    for(int i=0;i<rpos;i++) path[i] = res[rpos-1-i];
    *path_len = rpos;
    return 1;
}
// 说明：若为无向图，可把每条无向边表示为 work[u][v]++ 和 work[v][u]++，在访问时两边皆减 1（或在上层只访问 u<v 的边并特殊处理）。
```
